Suggests replacing JUnit4 ∧ JUnit5 assertions with AssertJ assertions.
Before (JUnit4 or 5) After (AssertJ)
assertEquals(e, a);
assertNotEquals(e, a); 		assertThat(a).isEqualTo(e);
assertThat(a).isNotEqualTo(e);
assertTrue(a);
assertFalse(a); 		assertThat(a).isTrue();
assertThat(a).isFalse();
assertNull(a);
assertNotNull(a); 		assertThat(a).isNull();
assertThat(a).isNotNull();
assertThrows(ex, () -> {}); assertThatThrownBy(() -> {}).isInstanceOf(ex);
assertEquals(e, a, "message"); (similar for others above when they take an explanation message) assertThat(a).describedAs("message").isEqualTo(e);
